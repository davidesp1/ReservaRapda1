let currentMenuPage = 1;
let menuPageSize = 10;
let menuCurrentSearch = '';
let menuCurrentCategory = '';
let menuCurrentStatus = '';
let paginatedMenu = [];

function filterMenuProdutos() {
    let filtered = menuProdutos.filter(p =&amp;gt; {
        let search = menuCurrentSearch.trim().toLowerCase();
        let matchSearch = !search ||
            p.nome.toLowerCase().includes(search) ||
            p.categoria.toLowerCase().includes(search);

        let matchCategory = !menuCurrentCategory || p.categoria === menuCurrentCategory;
        let matchStatus = !menuCurrentStatus || (menuCurrentStatus === 'disponivel' ? p.status === 'disponivel' : p.status === 'indisponivel');
        return matchSearch &amp;amp;&amp;amp; matchCategory &amp;amp;&amp;amp; matchStatus;
    });
    return filtered;
}

function renderMenuTable() {
    const tbody = document.getElementById('menu-tbody');
    tbody.innerHTML = '';

    let filtered = filterMenuProdutos();
    let total = filtered.length;
    let pageSize = menuPageSize === 'all' ? total : menuPageSize;
    let totalPages = pageSize === 0 ? 1 : Math.ceil(total / pageSize);
    if (currentMenuPage &amp;gt; totalPages) currentMenuPage = totalPages || 1;

    let start = ((currentMenuPage - 1) * pageSize);
    let end = menuPageSize === 'all' ? total : start + pageSize;
    paginatedMenu = filtered.slice(start, end);

    paginatedMenu.forEach(produto =&amp;gt; {
        let statusBadge = produto.status === 'disponivel'
            ? `&amp;lt;span class="inline-flex items-center px-2 py-1 text-xs font-semibold bg-green-100 text-brazil-green rounded"&amp;gt;&amp;lt;i class="fa-solid fa-circle mr-1 text-xs"&amp;gt;&amp;lt;/i&amp;gt; Disponível&amp;lt;/span&amp;gt;`
            : `&amp;lt;span class="inline-flex items-center px-2 py-1 text-xs font-semibold bg-red-100 text-brazil-red rounded"&amp;gt;&amp;lt;i class="fa-solid fa-circle mr-1 text-xs"&amp;gt;&amp;lt;/i&amp;gt; Indisponível&amp;lt;/span&amp;gt;`;

        tbody.innerHTML += `
            &amp;lt;tr class="hover:bg-gray-50 transition group"&amp;gt;
                &amp;lt;td class="px-6 py-4 flex items-center min-w-[220px]"&amp;gt;
                    &amp;lt;img src="${produto.foto}" alt="" class="w-12 h-12 rounded-lg object-cover border-2 border-brazil-yellow mr-3 shadow-sm"&amp;gt;
                    &amp;lt;div&amp;gt;
                        &amp;lt;div class="text-base font-semibold font-montserrat"&amp;gt;${produto.nome}&amp;lt;/div&amp;gt;
                    &amp;lt;/div&amp;gt;
                &amp;lt;/td&amp;gt;
                &amp;lt;td class="px-4 py-4 min-w-[90px]"&amp;gt;
                    &amp;lt;span class="text-brazil-blue font-bold"&amp;gt;R$ ${produto.preco.toFixed(2).replace('.',',')}&amp;lt;/span&amp;gt;
                &amp;lt;/td&amp;gt;
                &amp;lt;td class="px-4 py-4 min-w-[120px]"&amp;gt;
                    &amp;lt;span class="inline-flex items-center px-2 py-1 text-xs font-semibold bg-brazil-yellow bg-opacity-20 text-brazil-blue rounded"&amp;gt;${produto.categoria}&amp;lt;/span&amp;gt;
                &amp;lt;/td&amp;gt;
                &amp;lt;td class="px-4 py-4 text-center"&amp;gt;
                    &amp;lt;span class="text-gray-700"&amp;gt;${produto.stock}&amp;lt;/span&amp;gt;
                &amp;lt;/td&amp;gt;
                &amp;lt;td class="px-4 py-4 text-center"&amp;gt;
                    ${statusBadge}
                &amp;lt;/td&amp;gt;
                &amp;lt;td class="px-4 py-4 text-center"&amp;gt;
                    &amp;lt;button class="p-2 rounded-lg text-brazil-blue hover:bg-brazil-blue hover:text-white transition mr-1" title="Visualizar"&amp;gt;
                        &amp;lt;i class="fa-solid fa-eye"&amp;gt;&amp;lt;/i&amp;gt;
                    &amp;lt;/button&amp;gt;
                    &amp;lt;button class="p-2 rounded-lg text-brazil-green hover:bg-brazil-green hover:text-white transition mr-1" title="Editar"&amp;gt;
                        &amp;lt;i class="fa-solid fa-pen"&amp;gt;&amp;lt;/i&amp;gt;
                    &amp;lt;/button&amp;gt;
                    &amp;lt;button class="p-2 rounded-lg text-brazil-red hover:bg-brazil-red hover:text-white transition" title="Eliminar"&amp;gt;
                        &amp;lt;i class="fa-solid fa-trash"&amp;gt;&amp;lt;/i&amp;gt;
                    &amp;lt;/button&amp;gt;
                &amp;lt;/td&amp;gt;
            &amp;lt;/tr&amp;gt;
        `;
    });

    // Paginação info
    const info = document.getElementById('menu-pagination-info');
    let showingFrom = total === 0 ? 0 : (start + 1);
    let showingTo = end &amp;gt; total ? total : end;
    info.innerText = `Exibindo ${showingFrom}-${showingTo} de ${total} produtos`;

    // Paginação
    let pag = document.getElementById('menu-pagination');
    pag.innerHTML = '';
    if (menuPageSize === 'all' || totalPages &amp;lt;= 1) return;

    let prevDisabled = currentMenuPage === 1 ? 'opacity-50 pointer-events-none' : '';
    let nextDisabled = currentMenuPage === totalPages ? 'opacity-50 pointer-events-none' : '';

    pag.innerHTML += `&amp;lt;button class="px-2 py-1 rounded text-brazil-blue hover:bg-brazil-blue hover:text-white transition ${prevDisabled}" onclick="menuGotoPage(${currentMenuPage - 1})"&amp;gt;&amp;lt;i class="fa-solid fa-angle-left"&amp;gt;&amp;lt;/i&amp;gt;&amp;lt;/button&amp;gt;`;

    for (let i = 1; i &amp;lt;= totalPages; i++) {
        let active = i === currentMenuPage ? 'bg-brazil-yellow font-bold' : 'hover:bg-brazil-blue hover:text-white';
        pag.innerHTML += `&amp;lt;button class="px-2 py-1 rounded text-brazil-blue ${active} transition" onclick="menuGotoPage(${i})"&amp;gt;${i}&amp;lt;/button&amp;gt;`;
    }

    pag.innerHTML += `&amp;lt;button class="px-2 py-1 rounded text-brazil-blue hover:bg-brazil-blue hover:text-white transition ${nextDisabled}" onclick="menuGotoPage(${currentMenuPage + 1})"&amp;gt;&amp;lt;i class="fa-solid fa-angle-right"&amp;gt;&amp;lt;/i&amp;gt;&amp;lt;/button&amp;gt;`;
}

function menuGotoPage(page) {
    currentMenuPage = page;
    renderMenuTable();
}

document.getElementById('menu-search').addEventListener('input', e =&amp;gt; {
    menuCurrentSearch = e.target.value;
    currentMenuPage = 1;
    renderMenuTable();
});
document.getElementById('menu-category-filter').addEventListener('change', e =&amp;gt; {
    menuCurrentCategory = e.target.value;
    currentMenuPage = 1;
    renderMenuTable();
});
document.getElementById('menu-status-filter').addEventListener('change', e =&amp;gt; {
